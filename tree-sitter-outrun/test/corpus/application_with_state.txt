================
Application with configuration
================

struct ConfigApp() {
  def load_config(args: List<String>): AppConfig {
    AppConfig.from_args(args: args)
  }
}

impl Application for ConfigApp {
  def start(args: List<String>): Result<(), ApplicationError> {
    let config = ConfigApp.load_config(args: args)
    Logger.info(message: "Application configured")
    Server.run(config: config)
    Result.ok(value: ())
  }
}
