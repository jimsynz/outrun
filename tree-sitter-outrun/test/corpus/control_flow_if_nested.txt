def classify_number(x: Integer): String {
    if Integer.positive?(x) {
        if Integer.even?(x) {
            "positive even"
        } else {
            if x > 10 {
                "positive odd large"
            } else {
                "positive odd small"
            }
        }
    } else {
        if Integer.zero?(x) {
            "zero"
        } else {
            "negative"
        }
    }
}
